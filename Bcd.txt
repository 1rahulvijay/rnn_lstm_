WITH hub AS (
    -- Your CTE for hub table
    SELECT *
    FROM your_hub_source
)

SELECT
    hub.unit_code,
    COALESCE(hub_exceptions.cluster, hub.cluster) AS cluster,
    COALESCE(hub_exceptions.hub_region, hub.hub_region) AS hub_region,
    -- Include any other columns from hub or hub_exceptions as needed
    hub.column1,
    hub.column2,
    hub_exceptions.columnA
FROM hub
LEFT JOIN hub_exceptions
    ON hub.unit_code = hub_exceptions.unit_code
WHERE hub.db_name = 'abc_'
   OR hub_exceptions.db_name = 'abc_';
