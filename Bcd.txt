
SELECT 
    employee_name,
    hire_date,
    status,
    CASE 
        WHEN status = 'terminated' THEN 'Terminated'
        ELSE TO_CHAR(TRUNC(MONTHS_BETWEEN(SYSDATE, hire_date) / 12)) || ' years, ' ||
             TO_CHAR(TRUNC(MOD(MONTHS_BETWEEN(SYSDATE, hire_date), 12))) || ' months, ' ||
             TO_CHAR(SYSDATE - ADD_MONTHS(hire_date, TRUNC(MONTHS_BETWEEN(SYSDATE, hire_date)))) || ' days'
    END AS employment_duration
FROM 
    employee;
