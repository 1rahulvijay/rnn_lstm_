import pandas as pd

# Example DataFrame with new column 'D'
df = pd.DataFrame({
    'A': [1, 2, 3],
    'B': [4, 5, 6],
    'C': [7, 8, 9],
    'D': [10, 11, 12]  # This is a new column
})

# Predefined column order for known columns
desired_order = ['B', 'A']  # Desired order for B and A, but maintain positions of other columns

# Get the current columns of the DataFrame
current_columns = df.columns.tolist()

# Create a list for final ordered columns
final_column_order = []

# Iterate through current columns and add them in the desired order if in `desired_order`
# or retain their original positions if not in `desired_order`
for col in current_columns:
    if col in desired_order:
        # Add columns in the order they appear in `desired_order`
        desired_col = [dcol for dcol in desired_order if dcol == col][0]
        final_column_order.append(desired_col)
    else:
        # Maintain original position of any columns not in `desired_order`
        final_column_order.append(col)

# Reorder the DataFrame based on the final column order
df = df[final_column_order]

print(df)
