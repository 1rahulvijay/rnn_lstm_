SELECT 
    t1.status,
    t1.id,
    t1.unit,
    t1.id_version
FROM 
    your_table t1
JOIN (
    SELECT 
        id,
        MAX(id_version) AS max_version
    FROM 
        your_table
    GROUP BY 
        id
) latest 
    ON t1.id = latest.id 
    AND t1.id_version = latest.max_version
WHERE 
    t1.unit = 'Israel'
    AND NOT EXISTS (
        SELECT 
            1
        FROM 
            your_table t2
        WHERE 
            t2.id = t1.id
            AND t2.id_version = latest.max_version
            AND t2.unit <> 'Israel'
    )
ORDER BY 
    t1.id_version DESC;
