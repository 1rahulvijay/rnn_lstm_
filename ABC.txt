SELECT 
    e.employee_name AS employee_name,
    m1.employee_name AS manager_level_1,
    m2.employee_name AS manager_level_2,
    m3.employee_name AS manager_level_3,
    m4.employee_name AS manager_level_4,
    m5.employee_name AS manager_level_5,
    m6.employee_name AS manager_level_6
FROM 
    employee e
LEFT JOIN 
    manager mgr ON e.employee_id = mgr.employee_id
LEFT JOIN 
    employee m1 ON mgr.manager_id_1 = m1.employee_id
LEFT JOIN 
    employee m2 ON mgr.manager_id_2 = m2.employee_id
LEFT JOIN 
    employee m3 ON mgr.manager_id_3 = m3.employee_id
LEFT JOIN 
    employee m4 ON mgr.manager_id_4 = m4.employee_id
LEFT JOIN 
    employee m5 ON mgr.manager_id_5 = m5.employee_id
LEFT JOIN 
    employee m6 ON mgr.manager_id_6 = m6.employee_id;
