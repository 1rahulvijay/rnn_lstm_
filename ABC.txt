SELECT 
    e1.employee_name AS Employee,
    e1.manager_name AS Direct_Manager,
    e2.manager_name AS Manager_of_Direct_Manager,
    e3.manager_name AS Manager_of_Manager,
    e4.manager_name AS Manager_of_Manager_of_Manager,
    e5.manager_name AS Manager_of_Manager_of_Manager_of_Manager
FROM employees e1
LEFT JOIN employees e2 ON e1.manager_name = e2.employee_name
LEFT JOIN employees e3 ON e2.manager_name = e3.employee_name
LEFT JOIN employees e4 ON e3.manager_name = e4.employee_name
LEFT JOIN employees e5 ON e4.manager_name = e5.employee_name;
