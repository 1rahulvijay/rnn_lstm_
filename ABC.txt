SELECT 
    t1.status,
    t1.id,
    t1.unit,
    t1.id_version
FROM 
    your_table t1
INNER JOIN (
    SELECT 
        t2.id,
        MAX(t2.id_version) AS max_version
    FROM 
        your_table t2
    GROUP BY 
        t2.id
) latest 
    ON t1.id = latest.id 
    AND t1.id_version = latest.max_version
WHERE 
    t1.unit = 'Israel'
ORDER BY 
    t1.id_version DESC;
