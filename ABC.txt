import os
from datetime import datetime

def get_all_files_info(directory):
    files_info = {}

    for filename in os.listdir(directory):
        filepath = os.path.join(directory, filename)
        
        if os.path.isfile(filepath):
            mod_time = os.path.getmtime(filepath)
            mod_time = datetime.fromtimestamp(mod_time)
            files_info[filename] = mod_time

    return files_info

def check_files_exist(directory, file_list):
    # Get the dictionary with all files and their modification dates
    all_files_info = get_all_files_info(directory)
    
    # Check if all files in the list exist in the dictionary
    for file in file_list:
        if file not in all_files_info:
            print(f"Error: '{file}' does not exist in the directory.")
            # Exit the program
            raise FileNotFoundError(f"File '{file}' not found in directory.")
    
    print("All files exist.")

# Example usage:
directory = '/path/to/your/files'
files_to_check = ['file1.txt', 'file2.txt', 'file3.txt']

try:
    check_files_exist(directory, files_to_check)
except FileNotFoundError as e:
    print(e)
