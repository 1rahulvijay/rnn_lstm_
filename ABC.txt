SELECT 
    t1.status,
    t1.id,
    t1.unit,
    t1.id_version
FROM 
    your_table t1
JOIN (
    SELECT 
        id,
        MAX(id_version) AS max_version
    FROM 
        your_table
    GROUP BY 
        id
) latest 
    ON t1.id = latest.id 
    AND t1.id_version = latest.max_version
LEFT JOIN your_table t2 
    ON t1.id = t2.id 
    AND t1.id_version = t2.id_version 
    AND t2.unit <> 'Israel'
WHERE 
    t1.unit = 'Israel'
    AND t2.id IS NULL
ORDER BY 
    t1.id_version DESC;
